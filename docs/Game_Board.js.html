	var _params = params || {};

	this._columns = _params.columns || 3;
	this._rows = _params.rows || 3;

	this._slots = Array.apply(null, Array(this.size)).map(function() {

/**

 * @function isFull
 * @memberof Panda.TripleTriad.Game_Board.prototype
